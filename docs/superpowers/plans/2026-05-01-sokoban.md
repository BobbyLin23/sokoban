Create `src/game/gameEngine.ts`:

```ts
import type {
  Direction,
  GameSnapshot,
  GameState,
  LevelDefinition,
  ParsedLevel,
  Point,
  Tile,
} from './types'

const DELTAS: Record<Direction, Point> = {
  up: { row: -1, col: 0 },
  down: { row: 1, col: 0 },
  left: { row: 0, col: -1 },
  right: { row: 0, col: 1 },
}

const keyOf = (point: Point) => `${point.row}:${point.col}`

const samePoint = (a: Point, b: Point) => a.row === b.row && a.col === b.col

const addPoints = (a: Point, b: Point): Point => ({
  row: a.row + b.row,
  col: a.col + b.col,
})

const clonePoints = (points: Point[]) => points.map((point) => ({ ...point }))

const snapshotOf = (game: GameState): GameSnapshot => ({
  player: { ...game.player },
  crates: clonePoints(game.crates),
  moves: game.moves,
  won: game.won,
})

export const parseLevel = (definition: LevelDefinition): ParsedLevel => {
  if (definition.map.length === 0) {
    throw new Error(`Level ${definition.id} has no rows`)
  }

  const cols = definition.map[0].length
  const tiles: Tile[][] = []
  const targets: Point[] = []
  const initialCrates: Point[] = []
  let initialPlayer: Point | undefined

  definition.map.forEach((rowText, row) => {
    if (rowText.length !== cols) {
      throw new Error(`Level ${definition.id} must be rectangular`)
    }

    const tileRow: Tile[] = []

    Array.from(rowText).forEach((char, col) => {
      const point = { row, col }

      if (char === '#') {
        tileRow.push('wall')
        return
      }

      if (char === '.' || char === '*' || char === '+') {
        tileRow.push('target')
        targets.push(point)
      } else {
        tileRow.push('floor')
      }

      if (char === '$' || char === '*') {
        initialCrates.push(point)
      }

      if (char === '@' || char === '+') {
        if (initialPlayer) {
          throw new Error(`Level ${definition.id} has multiple players`)
        }
        initialPlayer = point
      }

      if (!'# .$*@+'.includes(char)) {
        throw new Error(`Level ${definition.id} contains invalid tile "${char}"`)
      }
    })

    tiles.push(tileRow)
  })

  if (!initialPlayer) {
    throw new Error(`Level ${definition.id} has no player`)
  }

  if (targets.length === 0 || initialCrates.length === 0) {
    throw new Error(`Level ${definition.id} needs targets and crates`)
  }

  return {
    id: definition.id,
    title: definition.title,
    rows: definition.map.length,
    cols,
    tiles,
    targets,
    initialPlayer,
    initialCrates,
  }
}

export const createGame = (level: ParsedLevel): GameState => ({
  level,
  player: { ...level.initialPlayer },
  crates: clonePoints(level.initialCrates),
  moves: 0,
  won: isWon(level, level.initialCrates),
  history: [],
})

export const isWon = (level: ParsedLevel, crates: Point[]) => {
  const crateKeys = new Set(crates.map(keyOf))
  return level.targets.every((target) => crateKeys.has(keyOf(target)))
}

export const getTile = (level: ParsedLevel, point: Point): Tile => {
  return level.tiles[point.row]?.[point.col] ?? 'wall'
}

export const hasCrate = (crates: Point[], point: Point) => {
  return crates.some((crate) => samePoint(crate, point))
}

export const movePlayer = (game: GameState, direction: Direction): GameState => {
  if (game.won) {
    return game
  }

  const delta = DELTAS[direction]
  const destination = addPoints(game.player, delta)

  if (getTile(game.level, destination) === 'wall') {
    return game
  }

  const crateIndex = game.crates.findIndex((crate) => samePoint(crate, destination))
  if (crateIndex >= 0) {
    const crateDestination = addPoints(destination, delta)
    if (getTile(game.level, crateDestination) === 'wall' || hasCrate(game.crates, crateDestination)) {
      return game
    }

    const crates = clonePoints(game.crates)
    crates[crateIndex] = crateDestination

    return {
      ...game,
      player: destination,
      crates,
      moves: game.moves + 1,
      won: isWon(game.level, crates),
      history: [...game.history, snapshotOf(game)],
    }
  }

  return {
    ...game,
    player: destination,
    moves: game.moves + 1,
    history: [...game.history, snapshotOf(game)],
  }
}

export const undoMove = (game: GameState): GameState => {
  const previous = game.history.at(-1)
  if (!previous) {
    return game
  }

  return {
    ...game,
    ...previous,
    crates: clonePoints(previous.crates),
    player: { ...previous.player },
    history: game.history.slice(0, -1),
  }
}

export const restartGame = (game: GameState): GameState => createGame(game.level)
```

```ts
import type { BestResult, ProgressState } from './types'

const STORAGE_KEY = 'sokoban-progress-v1'

export const defaultProgress: ProgressState = {
  highestUnlockedLevel: 1,
  selectedLevel: 1,
  soundEnabled: true,
  bestResults: {},
}

const normalizeProgress = (value: Partial<ProgressState>): ProgressState => ({
  highestUnlockedLevel:
    typeof value.highestUnlockedLevel === 'number' && value.highestUnlockedLevel >= 1
      ? value.highestUnlockedLevel
      : defaultProgress.highestUnlockedLevel,
  selectedLevel:
    typeof value.selectedLevel === 'number' && value.selectedLevel >= 1
      ? value.selectedLevel
      : defaultProgress.selectedLevel,
  soundEnabled:
    typeof value.soundEnabled === 'boolean'
      ? value.soundEnabled
      : defaultProgress.soundEnabled,
  bestResults:
    value.bestResults && typeof value.bestResults === 'object'
      ? value.bestResults
      : defaultProgress.bestResults,
})

export const loadProgress = (): ProgressState => {
  try {
    const raw = window.localStorage.getItem(STORAGE_KEY)
    if (!raw) {
      return defaultProgress
    }
    return normalizeProgress(JSON.parse(raw) as Partial<ProgressState>)
  } catch {
    return defaultProgress
  }
}

export const saveProgress = (progress: ProgressState) => {
  try {
    window.localStorage.setItem(STORAGE_KEY, JSON.stringify(progress))
  } catch {
    // Saving progress is best-effort; gameplay should continue if storage is unavailable.
  }
}

export const withCompletedLevel = (
  progress: ProgressState,
  levelId: number,
  elapsedSeconds: number,
  moves: number,
  maxLevel: number,
): ProgressState => {
  const previousBest = progress.bestResults[levelId]
  const nextBest: BestResult =
    !previousBest || moves < previousBest.moves || (moves === previousBest.moves && elapsedSeconds < previousBest.seconds)
      ? { moves, seconds: elapsedSeconds }
      : previousBest

  return {
    ...progress,
    highestUnlockedLevel: Math.min(maxLevel, Math.max(progress.highestUnlockedLevel, levelId + 1)),
    bestResults: {
      ...progress.bestResults,
      [levelId]: nextBest,
    },
  }
}
```

```tsx
type BoardProps = {
  game: GameState

const pointKey = (point: Point) => `${point.row}:${point.col}`

export function Board({ game }: BoardProps) {
  const targetKeys = new Set(game.level.targets.map(pointKey))

  return (
    <div
      className="grid max-h-[min(68svh,720px)] max-w-[min(92vw,720px)] touch-none gap-1 rounded-2xl border border-cyan-400/20 bg-slate-950/80 p-2 shadow-[0_0_40px_rgba(34,211,238,0.16)]"

        gridTemplateColumns: `repeat(${game.level.cols}, minmax(0, 1fr))`,
        aspectRatio: `${game.level.cols} / ${game.level.rows}`,
      }}
      aria-label={`Level ${game.level.id}: ${game.level.title}`}
    >
      {Array.from({ length: game.level.rows * game.level.cols }, (_, index) => {
        const point = { row: Math.floor(index / game.level.cols), col: index % game.level.cols }
        const tile = getTile(game.level, point)
        const crate = hasCrate(game.crates, point)
        const target = targetKeys.has(pointKey(point))
        const player = game.player.row === point.row && game.player.col === point.col

        const className = [
          'relative aspect-square rounded-md border transition-transform duration-150',
          tile === 'wall' ? 'border-slate-500/30 bg-slate-700 shadow-inner' : 'border-slate-800/80 bg-slate-900',
```

Replace `src/App.tsx`:

```tsx
import { useCallback, useEffect, useMemo, useState } from 'react'
import './App.css'
import { Board } from './components/Board'
import { Controls } from './components/Controls'
import { Hud } from './components/Hud'
import { LevelSelect } from './components/LevelSelect'
import { WinDialog } from './components/WinDialog'
import { createGame, movePlayer, parseLevel, restartGame, undoMove } from './game/gameEngine'
import { LEVELS } from './game/levels'
import { loadProgress, saveProgress, withCompletedLevel } from './game/progress'
import type { Direction, GameState, ProgressState } from './game/types'

const directionFromKey = (key: string): Direction | undefined => {
  const normalized = key.toLowerCase()
  if (normalized === 'arrowup' || normalized === 'w') return 'up'
  if (normalized === 'arrowdown' || normalized === 's') return 'down'
  if (normalized === 'arrowleft' || normalized === 'a') return 'left'
  if (normalized === 'arrowright' || normalized === 'd') return 'right'
  return undefined
}

const createGameByLevelId = (levelId: number): GameState => {
  const level = LEVELS.find((item) => item.id === levelId) ?? LEVELS[0]
  return createGame(parseLevel(level))
}

function App() {
  const [progress, setProgress] = useState<ProgressState>(() => loadProgress())
  const [game, setGame] = useState<GameState>(() => createGameByLevelId(loadProgress().selectedLevel))
  const [levelSelectOpen, setLevelSelectOpen] = useState(false)
  const [elapsedSeconds, setElapsedSeconds] = useState(0)
  const [startedAt, setStartedAt] = useState(() => Date.now())
  const [completionRecorded, setCompletionRecorded] = useState(false)

  const currentBest = progress.bestResults[game.level.id]

  const setLevel = useCallback((levelId: number) => {
    if (levelId > progress.highestUnlockedLevel) {
      return
    }

    const nextGame = createGameByLevelId(levelId)
    setGame(nextGame)
    setElapsedSeconds(0)
    setStartedAt(Date.now())
    setCompletionRecorded(false)
    setLevelSelectOpen(false)
    setProgress((current) => ({ ...current, selectedLevel: levelId }))
  }, [progress.highestUnlockedLevel])

  const handleMove = useCallback((direction: Direction) => {
    setGame((current) => {
      const next = movePlayer(current, direction)
      return next
    })
  }, [])

  const handleRestart = useCallback(() => {
    setGame((current) => restartGame(current))
    setElapsedSeconds(0)
    setStartedAt(Date.now())
    setCompletionRecorded(false)
  }, [])

  const handleUndo = useCallback(() => {
    setGame((current) => undoMove(current))
  }, [])

  useEffect(() => {
    saveProgress(progress)
  }, [progress])

  useEffect(() => {
    if (game.won) {
      return
    }

    const timer = window.setInterval(() => {
      setElapsedSeconds(Math.floor((Date.now() - startedAt) / 1000))
    }, 500)

    return () => window.clearInterval(timer)
  }, [game.won, startedAt])

  useEffect(() => {
    if (!game.won || completionRecorded) {
      return
    }

    setCompletionRecorded(true)
    setProgress((current) =>
      withCompletedLevel(current, game.level.id, elapsedSeconds, game.moves, LEVELS.length),
    )
  }, [completionRecorded, elapsedSeconds, game.level.id, game.moves, game.won])

  useEffect(() => {
    const onKeyDown = (event: KeyboardEvent) => {
      const direction = directionFromKey(event.key)
      if (!direction) {
        return
      }
      event.preventDefault()
      handleMove(direction)
    }

    window.addEventListener('keydown', onKeyDown, { passive: false })
    return () => window.removeEventListener('keydown', onKeyDown)
  }, [handleMove])

  const isLastLevel = game.level.id === LEVELS.length
  const nextLevel = useMemo(() => Math.min(game.level.id + 1, LEVELS.length), [game.level.id])

  return (
    <main className="min-h-svh bg-[radial-gradient(circle_at_top,#14264d_0,#070a12_48%,#020617_100%)] px-4 py-4 text-slate-100 sm:px-6 lg:px-8">
      <div className="mx-auto flex min-h-[calc(100svh-2rem)] w-full max-w-6xl flex-col gap-5">
        <Hud
          levelTitle={game.level.title}
          levelId={game.level.id}
          moves={game.moves}
          seconds={elapsedSeconds}
          best={currentBest}
          soundEnabled={progress.soundEnabled}
          onToggleSound={() => setProgress((current) => ({ ...current, soundEnabled: !current.soundEnabled }))}
          onOpenLevels={() => setLevelSelectOpen(true)}
        />

        <section className="flex flex-1 flex-col items-center justify-center gap-6 lg:flex-row lg:items-center">
          <Board game={game} />
          <Controls canUndo={game.history.length > 0} onMove={handleMove} onUndo={handleUndo} onRestart={handleRestart} />
        </section>
      </div>

      <LevelSelect
        open={levelSelectOpen}
        levels={LEVELS}
        currentLevel={game.level.id}
        highestUnlockedLevel={progress.highestUnlockedLevel}
        bestResults={progress.bestResults}
        onSelect={setLevel}
        onClose={() => setLevelSelectOpen(false)}
      />

      <WinDialog
        open={game.won}
        levelId={game.level.id}
        isLastLevel={isLastLevel}
        moves={game.moves}
        seconds={elapsedSeconds}
        onNext={() => setLevel(nextLevel)}
        onRetry={handleRestart}
        onLevels={() => setLevelSelectOpen(true)}
      />
    </main>
  )
}

export default App
```

```ts
const playTone = (enabled: boolean, frequency: number, duration = 0.06) => {
  if (!enabled) {
    return
  }

  const AudioContextClass = window.AudioContext || window.webkitAudioContext
  if (!AudioContextClass) {
    return
  }

  const context = new AudioContextClass()
  const oscillator = context.createOscillator()
  const gain = context.createGain()

  oscillator.frequency.value = frequency
  oscillator.type = 'square'
  gain.gain.value = 0.025
  oscillator.connect(gain)
  gain.connect(context.destination)
  oscillator.start()
  oscillator.stop(context.currentTime + duration)
}
```
